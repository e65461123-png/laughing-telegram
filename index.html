// إضافة رادار القيادة في الخلفية function drawRadar() { ctx.strokeStyle = 'rgba(0, 255, 247, 0.2)'; ctx.lineWidth = 2; ctx.beginPath(); ctx.arc(canvas.width/2, canvas.height/2, 200, 0, Math.PI*2); ctx.stroke(); // خط المسح الدوار let angle = (Date.now() / 1000) % (Math.PI * 2); ctx.moveTo(canvas.width/2, canvas.height/2); ctx.lineTo(canvas.width/2 + Math.cos(angle)*200, canvas.height/2 + Math.sin(angle)*200); ctx.stroke(); } // تعديل وظيفة التحريك لتشمل الرادار function animate() { ctx.clearRect(0, 0, canvas.width, canvas.height); drawRadar(); // تفعيل الرادار parts.forEach(p => { p.update(); p.draw(); }); requestAnimationFrame(animate); }
