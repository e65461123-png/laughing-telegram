""" self.wfile.write(html_content.encode('utf-8')) def do_POST(self): if self.path == '/update': content_length = int(self.headers['Content-Length']) post_data = self.rfile.read(content_length).decode('utf-8') params = urllib.parse.parse_qs(post_data) try: user_check = params.get('user')[0] pass_check = params.get('pass')[0] if user_check == 'Essam' and pass_check == 'Essam369': server_data["initiative_text"] = params.get('new_init')[0] server_data["news"] = params.get('new_news')[0] except: pass self.send_response(303) self.send_header('Location', '/') self.end_headers() with socketserver.TCPServer(("", PORT), AtherProtocolServer) as httpd: print(f"[✔] تم تشغيل السيرفر الآمن والمصفح بالكامل") webbrowser.open(f"http://localhost:8080") httpd.serve_forever()
